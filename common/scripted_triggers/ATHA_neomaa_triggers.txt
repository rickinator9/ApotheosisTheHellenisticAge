can_recruit_elite_maa = {
  is_independent_ruler = yes
}

can_recruit_greek_maa_trigger = {
  OR = {
    culture = { has_cultural_pillar = heritage_greek }
    any_realm_province = {
      culture = { has_cultural_pillar = heritage_greek }
    }
    
    # Exceptions for Mathian Illyrian culture since they were quite Hellenised.
    has_culture_or_parent_culture = { CULTURE = culture:mathian }
    any_realm_province = {
      has_culture_or_parent_culture = { CULTURE = culture:mathian }
    }
  }
}

can_recruit_cretan_maa_trigger = {
  OR = {
    has_culture_or_parent_culture = { CULTURE = culture:cretan }
    any_realm_province = {
      OR = {
        has_culture_or_parent_culture = { CULTURE = culture:cretan }
        geographical_region = ATHA_aor_cretan # Cretans kept being famed archers into medieval times, so even if Cretan culture is completely eradicated they are still recruitable from Crete.
      }
    }
  }
}

can_recruit_maa_agrianikoi_pelekephoroi_trigger = {
  OR = {
    has_culture_or_parent_culture = { CULTURE = culture:agrianian }
    AND = { # If you are a ruler of a Agrianian-adjacent culture ruling over an Agrianian-adjacent culture in the Agrianian regions, you can recruit their MaA as well.
      culture = {
        OR = {
          has_cultural_pillar = heritage_paeonian
          has_cultural_pillar = heritage_illyrian
          has_cultural_pillar = heritage_thracian
        }
      }
      any_realm_province = {
        culture = {
          OR = {
            has_cultural_pillar = heritage_paeonian
            has_cultural_pillar = heritage_illyrian
            has_cultural_pillar = heritage_thracian
          }
        }
        geographical_region = ATHA_aor_agrianes
      }
    }
    any_realm_province = {
      has_culture_or_parent_culture = { CULTURE = culture:agrianian }
    }
  }
}
