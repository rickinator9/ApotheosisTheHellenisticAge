###civilized#skirmishers###

can_recruit_maa_euzonoi_natively = {
  culture = { has_cultural_pillar = heritage_greek }
  NOR = {
    can_recruit_maa_kretikoi_peltastai_natively_trigger = yes
    can_recruit_maa_makedonikoi_peltastai_natively_trigger = yes
  }
}

can_recruit_maa_euzonoi_as_foreigner = {
  NOT = { can_recruit_maa_euzonoi_natively = yes }
  any_realm_province = {
    can_recruit_maa_euzonoi_natively = yes
  }
}

can_recruit_maa_kretikoi_peltastai_trigger = {
  OR = {
  culture = culture:cretan
  any_parent_culture_or_above = { this = culture:cretan }
  }
}

can_recruit_maa_kretikoi_peltastai_as_foreigner_trigger = {
  NOT = { can_recruit_maa_kretikoi_peltastai_natively_trigger = yes }
  any_realm_province = {
    can_recruit_maa_kretikoi_peltastai_natively_trigger = yes
  }
}

can_recruit_maa_makedonikoi_peltastai_trigger = {
  OR = {
  culture = culture:macedonian
  any_parent_culture_or_above = { this = culture:macedonian }
  }
}

can_recruit_maa_makedonikoi_peltastai_as_foreigner_trigger = {
  NOT = { can_recruit_maa_makedonikoi_peltastai_natively_trigger = yes }
  any_realm_province = {
    can_recruit_maa_makedonikoi_peltastai_natively_trigger = yes
  }
}
