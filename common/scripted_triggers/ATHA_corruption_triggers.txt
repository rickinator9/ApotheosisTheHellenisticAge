knows_about_character_being_corrupt_trigger = {
	OR = {
		$CHARACTER$ = { has_trait = corrupt }
		any_known_secret = {
			secret_owner = $CHARACTER$
			secret_type = secret_corrupt
		}
	}
}

is_corrupt_trigger = {
	OR = {
		has_trait = corrupt
		any_secret = {
			secret_type = secret_corrupt
		}
	}
}

secret_corrupt_is_valid_trigger = {
	$OWNER$ = {
		NOT = {
			has_trait = corrupt
		}
	}
}

secret_corrupt_is_criminal_trigger = {
	$OWNER$ = {
		OR = {
			faith = { has_doctrine_parameter = corruption_illegal }
			any_liege_or_above = { faith = { has_doctrine_parameter = corruption_illegal } }
		}
	}
}
