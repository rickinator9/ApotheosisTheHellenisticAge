		any_liege_or_above = {

				government_has_flag = government_is_clan
			}
		}
		any_vassal_or_below = {
			OR = {
				government_has_flag = government_is_feudal
				government_has_flag = government_is_clan
			}
		}
		any_neighboring_top_liege_realm_owner = {
			OR = {
				government_has_flag = government_is_feudal
				government_has_flag = government_is_clan
			}
			tier_difference = {
				target = prev
				value < -1
			}
		}
	}
}
