portrait_ATHA_spartan_clothing_trigger = {
	culture = culture:laconian
	OR = {
		has_government = city_state_spartan_government
		any_liege_or_above = { has_government = city_state_spartan_government }
	}
}

portrait_ATHA_hellenistic_greek_clothing_trigger = {
	OR = {
		culture = culture:macedonian
		culture = culture:epirote
	}
}

portrait_ATHA_classical_greek_clothing_trigger = {
	culture = { has_cultural_pillar = heritage_greek }
	NOR = {
		portrait_ATHA_spartan_clothing_trigger = yes
		portrait_ATHA_hellenistic_greek_clothing_trigger = yes
	}
}

portrait_ATHA_greek_clothing_trigger = {

}
