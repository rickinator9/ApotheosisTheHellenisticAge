# Wars / Battles

net_battle_wins = {
	value = 0
	if = {
		limit = { has_variable = battle_wins }
		add = var:battle_wins
	}
	if = {
		limit = { has_variable = battle_losses }
		subtract = var:battle_losses
	}
}

net_war_wins = {
	value = 0
	if = {
		limit = { has_variable = war_wins }
		add = var:war_wins
	}
	if = {
		limit = { has_variable = war_losses }
		subtract = var:war_losses
	}
}
