corruption_miniscule_increase = {
	value = 1
	if = {
		limit = {
			OR = {
				has_government = city_state_aristocratic_government
				has_government = city_state_oligarchic_government
				has_government = city_state_spartan_government
			}
		}
		multiply = 1.25
	}
	else_if = {
		limit = {
			has_government = city_state_tyrannical_government
		}
		multiply = 1.5
	}
	if = {
		limit = {
			has_trait = corrupt
		}
		multiply = 1.5
	}
}

#provide vice scripted values
provide_vice_cooldown_value = {
	value = 1
	if = {
		limit = {
			scope:root = {
				intrigue <= mediocre_skill_rating
			}
		}
		add = 1
	}
	if = {
		limit = {
			scope:root = {
				intrigue >= very_high_skill_rating
			}
		}
		add = -0.5
	}
}
