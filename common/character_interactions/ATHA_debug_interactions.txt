debug_make_tyranny = {
	category = interaction_debug_main
	use_diplomatic_range = no
	is_shown = {
		debug_only = yes
		NOT = { scope:recipient = scope:actor }
		scope:recipient = {
			has_government = city_state_democratic_government
		}
	}
	
	on_accept = {
		scope:recipient = {
			change_government_form = { GOVERNMENT = city_state_tyrannical_government }
		}
	}
	
	auto_accept = yes
}

debug_add_emergency = {
	category = interaction_debug_main
	use_diplomatic_range = no
	is_shown = {
		debug_only = yes
		scope:recipient = {
			has_government = city_state_democratic_government
			exists = cp:councillor_marshal
		}
	}

	on_accept = {
		scope:recipient = {
			cp:councillor_marshal = {
				trigger_event = { id = city_state_emergency.0001 }
			}
		}
	}
	
	auto_accept = yes
}

debug_abdicate = {
	category = interaction_debug_main
	use_diplomatic_range = no
	is_shown = {
		debug_only = yes
		scope:recipient = scope:actor
	}
	
	on_accept = {
		scope:recipient = {
			depose = yes
		}
	}
	
	auto_accept = yes
}
