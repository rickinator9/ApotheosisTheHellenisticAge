tradition_ceremonial_generosity = {
	category = societal

	layers = {
		0 = diplo
		1 = default_todo.dds
		2 = "rec2 copy 2.dds"
		3 = 6.dds
		4 = conversation.dds
	}

	parameters = {
		send_gift_prestige_gain = yes
		send_gift_double_gift = yes
	}

	cost = {
		prestige = {
			add = {
				value = tradition_base_cost
				desc = BASE
				format = "BASE_VALUE_FORMAT"
			}
			if = {
				limit = {
					NOT = {
						any_in_list = { list = traits this = flag:ethos_courtly }
					}
				}
				add = {
					value = tradition_incompatible_ethos_penalty
					desc = not_courtly_desc
				}
			}
			if = {
				limit = {
					trigger_if = {
						limit = {
							scope:character = {
								is_ai = no
							}
						}
						NOT = {
							any_ruler = {
								count >= 5
								culture = prev
								primary_title.tier >= tier_county
								has_trait = generous
							}
						}
					}
					trigger_else = {
						always = no
					}
				}
				add = {
					value = tradition_unfulfilled_criteria_penalty
					desc = rulers_with_generous_desc
				}
			}
			if = {
				limit = {
					trigger_if = {
						limit = {
							scope:character = {
								is_ai = no
							}
						}
						NOT = {
							any_ruler = {
								count >= 5
								culture = prev
								primary_title.tier >= tier_county
								has_trait = education_diplomacy_4
							}
						}
					}
					trigger_else = {
						always = no
					}
				}
				add = {
					value = tradition_unfulfilled_criteria_penalty
					desc = rulers_with_diplomacy_edu_desc
				}
			}
			
			multiply = tradition_replacement_cost_if_relevant
		}
	}
	ai_will_do = {
		value = 100
		if = {
			limit = {
				OR = {
					any_in_list = { list = traits this = flag:tradition_esteemed_hospitality }
					any_in_list = { list = traits this = flag:tradition_welcoming }
					any_in_list = { list = traits this = flag:tradition_charitable }
					any_in_list = { list = traits this = flag:tradition_ruling_caste }
					any_in_list = { list = traits this = flag:tradition_festivities }
				}
			}
			multiply = 0.1
		}
	}
}
