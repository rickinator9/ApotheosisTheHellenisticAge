language_greek = {
	type = language
	is_shown = {
		language_is_shown_trigger = {
			LANGUAGE = language_greek
		}
	}
	ai_will_do = {
		value = 10
		if = {
			limit = { has_cultural_pillar = language_greek }
			multiply = 10
		}
	}
	
	color = dorian
}

language_iranian = {
	type = language
	is_shown = {
		language_is_shown_trigger = {
			LANGUAGE = language_iranian
		}
	}
	ai_will_do = {
		value = 10
		if = {
			limit = { has_cultural_pillar = language_iranian }
			multiply = 10
		}
		if = {
			limit = {
				scope:character = {
					religion = religion:zoroastrianism_religion
				}
			}
			multiply = 2
		}
	}
	
	color = persian
}
