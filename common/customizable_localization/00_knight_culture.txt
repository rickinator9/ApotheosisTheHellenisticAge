KnightCulture = {
	type = character

	text = {
		trigger = {
			culture = { has_cultural_pillar = language_thracian }
		}
		localization_key = knight_zibythides
	}
	text = {
		trigger = {
			culture = { has_name_list = name_list_laconian }
			OR = { 
				has_government = city_state_spartan_government
				AND = { 
					has_government = hegemonic_league_government
					OR = {
						dynasty = dynasty:spartandyn001
						dynasty = dynasty:spartandyn002
					}
				}
			}
		}
		localization_key = knight_hippeus
	}
	text = { # Epirotes and Macedonians use Somatophylakes.
		trigger = {
			OR = {
				culture = { has_name_list = name_list_macedonian }
				culture = { has_name_list = name_list_epirote }
			}
		}
		localization_key = knight_somatophylax
	}
	text = {
		trigger = {
			OR = {
			# Doric
				culture = { has_name_list = name_list_achaean }
				culture = { has_name_list = name_list_cretan }
				culture = { has_name_list = name_list_dorian }
				culture = { has_name_list = name_list_dorian_northwest }
				culture = { has_name_list = name_list_laconian }
				culture = { has_name_list = name_list_messenian }
			# Aeolic
				culture = { has_name_list = name_list_aeolian }
				culture = { has_name_list = name_list_boeotian }
				culture = { has_name_list = name_list_thessalian }
			# Arcado-Cypriot
				culture = { has_name_list = name_list_arcadian }
			}
		}
		localization_key = knight_epilektos_hoplitas
	}
	text = {
		trigger = {
			culture = { has_cultural_pillar = language_greek }
		}
		localization_key = knight_epilektos_hoplites
	}
	text = {
		trigger = {
			culture = { has_cultural_pillar = language_celtic }
		}
		localization_key = knight_champion_no_tooltip
	}
	text = {
		localization_key = knight_default
		fallback = yes
	}
}
