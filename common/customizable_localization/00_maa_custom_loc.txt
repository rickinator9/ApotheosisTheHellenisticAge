GetRegionalInfantryMAA = {
	type = character
	random_valid = yes
	
	text = {
		trigger = {
			culture = { has_cultural_parameter = unlock_maa_spartiatai_hoplitai }
			OR = { 
				government_has_flag = city_state_spartan_government
				AND = { 
					government_has_flag = government_is_hegemonic_league
					OR = {
						dynasty = dynasty:spartandyn001
						dynasty = dynasty:spartandyn002
					}
				}
			}
		}
		localization_key = spartiatai_hoplitai
	}
	text = {
		trigger = { culture = { has_innovation = innovation_sarissa } }
		localization_key = pezhetairoi
	}
	text = {
		trigger = { culture = { has_cultural_parameter = unlock_maa_hoplitai } }
		localization_key = hoplitai
	}
	text = {
		trigger = { always = no }
		fallback = yes
		localization_key = infantry
	}
}
