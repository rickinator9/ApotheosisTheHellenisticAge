	type = phalanx

	damage = 32
	toughness = 20
	pursuit = 0
	screen = 0

	# Macedonian phalanx only really works well in flat terrain due to the tight formation.
	terrain_bonus = {
		hills = { damage = -5 }
		mountains = { damage = -10 }
		desert_mountains = { damage = -10 }
		wetlands = { damage = -15 }
		forest = { damage = -15 }
		taiga = { damage = -15 }

	}

	# Due to the lack of flexibility, Phalanxes do not counter any MaA.

	}

	buy_cost = { gold = phalanx_recruitment_cost }
	low_maintenance_cost = { gold = phalanx_low_maint_cost }
	high_maintenance_cost = { gold = phalanx_high_maint_cost }
	
	mercenary_fallback = yes
	holy_order_fallback = yes
	
	stack = 128 # Half the size of a syntagma.
	ai_quality = { value = @cultural_maa_extra_ai_score }
	icon = pikemen
}

# Traditional hoplite phalanx.
# Worse overall than Macedonian-style phalanx, but is more compeditive in hills and better armored.
hoplitai = {
	type = phalanx

	damage = 28
	toughness = 25
	pursuit = 3
	screen = 0

	terrain_bonus = {
		mountains = { damage = -5 }
		desert_mountains = { damage = -5 }
		wetlands = { damage = -10 }
		forest = { damage = -10 }
		taiga = { damage = -10 }
		jungle = { damage = -10 }
	}

	# Due to the lack of flexibility, Phalanxes do not counter any MaA.
	counters = {
	}

	buy_cost = { gold = phalanx_recruitment_cost }
	low_maintenance_cost = { gold = phalanx_low_maint_cost }
	high_maintenance_cost = { gold = phalanx_high_maint_cost }
	
	mercenary_fallback = yes
	holy_order_fallback = yes
	
	stack = 100
	ai_quality = {
		value = @cultural_maa_extra_ai_score

		if = {
			# Spartans love hoplites.
			limit = {
				is_spartan = yes
			}

			add = 40
		}

		if = {
			# City States love hoplites a whole lot more.
			limit = {
				government_has_flag = is_city_state_government
			}

			add = 40
		}
	
		# If Hoplites are clearly obsolete, prefer Phalangites.
		if = {
			limit = {
				culture = { has_innovation = innovation_sarissa }
				OR = {
					culture = culture:macedonian
					culture = culture:epirote
					current_year = 3514 # 240 BC
				}
			}

			add = -50
		}
	}
	icon = pikemen
}
