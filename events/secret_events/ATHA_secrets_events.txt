#Events for having secrets

namespace = ATHA_secrets

scripted_trigger secrets_0119_dislikes_the_corrupt_trigger = {
	OR = {
		has_trait = incorruptible
		NOR = {
			has_trait = corrupt
			this.num_virtuous_traits < this.num_sinful_traits
		}
	}
}
