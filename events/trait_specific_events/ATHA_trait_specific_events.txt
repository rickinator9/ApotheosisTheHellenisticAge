#A quick solution to a messy problem
ATHA_trait_specific.0001 = { #By Grathalas
	type = character_event
	title = ATHA_trait_specific.0001.t #write this
	desc = ATHA_trait_specific.0001.desc #write this

	theme = realm
	left_portrait = {
		character = scope:land_owner
		animation = beg
	}

	trigger = {
		is_corrupt_trigger = no
		is_imprisoned = no
		age >= 16
		NOR = {
			has_trait = incorruptible
		}

		any_vassal = {
			is_corrupt_trigger = yes
		}
	}

	immediate = {
		create_character = {
			location = root.capital_province
			template = merchant_template
			save_scope_as = land_owner
		}
		
		if = {
			limit = {
				any_vassal = {
					is_corrupt_trigger = yes
				}
			}
			random_vassal = {
				limit = {
					is_corrupt_trigger = yes
				}
				save_scope_as = corrupt_vassal
				if = {
					limit = {
						any_secret = {
							secret_type = secret_corrupt
							is_known_by = root
						}
					}
					save_scope_value_as = { #For desc purposes
						name = secret_known
						value = yes
					}
				}
				if = { #To save the secret so it can be revealed
					limit = {
						any_secret = {
							secret_type = secret_corrupt
							NOT = { is_known_by = root }
						}
					}
					random_secret = {
						limit = {
							secret_type = secret_corrupt
							NOT = { is_known_by = root }
						}
						save_scope_as = vassal_secret
						reveal_to = root
					}
				}
			}
		}
	}

	option = { #Summon the vassal
		name = ATHA_trait_specific.0001.a #write this

		trigger_event = {
			id = ATHA_trait_specific.0002
			days = { 1 5 }
		}
	}

	option = { #Dismiss the land owner
		name = ATHA_trait_specific.0001.b #write this

		stress_impact = {
			just = medium_stress_impact_gain
		}

		scope:corrupt_vassal = {
			add_opinion = {
				target = root
				modifier = pleased_opinion
				opinion = 10
			}
		}

		change_city_state_corruption = { AMOUNT_ARG = corruption_minor_increase }
	}
}

ATHA_trait_specific.0003 = {
	type = character_event
	title = ATHA_trait_specific.0003.t #write loc
	desc = ATHA_trait_specific.0003.desc #write loc

	theme = corruption
	override_background = {
		event_background = corridor_night
	}
	left_portrait = {
		character = scope:corrupt_vassal
		animation = scheme
	}

	option = { #Accept the deal
		name = ATHA_trait_specific.0003.a #write loc
		if = {
			limit = {
				is_corrupt_trigger = no
			}
			give_corrupt_secret_or_trait_effect = yes
		}
		scope:corrupt_vassal = {
			add_opinion = {
				target = root
				modifier = pleased_opinion
				opinion = 20
			}
			pay_short_term_gold = {
				gold = medium_gold_value
				target = root
			}
		}
		any_secret = {
			secret_type = secret_corrupt
			reveal_to = scope:corrupt_vassal
		}
		stress_impact = {
			just = massive_stress_impact_gain
		}
		change_city_state_corruption = { AMOUNT_ARG = corruption_major_increase }
	}

	option = { #Expose the vassal
		name = ATHA_trait_specific.0003.b #write loc
		scope:corrupt_vassal = {
			add_opinion = {
				target = root
				modifier = refusal_opinion
				opinion = -20
			}
		}
		scope:vassal_secret = {
			expose_secret = root
		}
		if = {
			limit = {
				NOT = {has_trait = incorruptible }
			}
			add_trait = incorruptible
		}
		stress_impact = {
			arbitrary = massive_stress_impact_gain
		}
		change_city_state_corruption = { AMOUNT_ARG = corruption_medium_decrease }
	}

	option = { #Decline the deal
		name = ATHA_trait_specific.0003.c #write loc
		scope:corrupt_vassal = {
			add_opinion = {
				target = root
				modifier = refusal_opinion
				opinion = -15
			}
		}
		trigger_event = {
			id = ATHA_trait_specific.0004
			days = { 1 3 }
		}
		stress_impact = {
			arbitrary = major_stress_impact_gain
		}
	}
}
