# We can use these modifiers for adding specific clothes to historical and vanity characters

clothes_scripted_characters = {

    usage = game
    selection_behavior = weighted_random

	greek_war = {
		dna_modifiers = {
			accessory = {
				mode = add
				gene = clothes
				template = greek_war_nobility_clothes
				range = { 0 1 } # For the randomness to work correctly
            }
        }   
        weight = {
            base = 0
			modifier = {
                add = 200
				exists = this
                exists = character:macedonian03948 # Alexander the Great
                this = character:macedonian03948 # Alexander the Great
            }
        }
    }

}
